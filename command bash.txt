git init - создать git repository

cat - просмотреть файл

git status - статус индексации файла

git add name.format - индексация файла

git add . - добавить в индекс все, что в папке

git restore --stage area - убрать из зоны индексирования

git commit -m "Name commit" - создать коммит

git log - проверить списки коммитов

git push origin namegit (имя ветки). - запушить коммит

git reset <key> <adress> - key --soft, --hard - удалить коммит

git remote add origin - добавляет ссылку 

git remote remove origin - удаляет репоситорий

--------------------------------------------
mkdir name file - создать папку

rm -r name - удалить папку или файл

touch - создать файл

cd - переход по папкам (просто cd переводит в самое начало)

cd .. - предыдущая папка

cd name - переход в папку (имя папки)

ls - просмотреть содержимое папки

nano - блокнот в терминале

ls -a - посмотеть папки, в том числе скрытые

mv name file new name file - переименовать файл

git clone namerep (new name) - скачать копию

git branch - ветки

git branch name - создание ветки

git switch name - переключаться между ветками

echo (...) - напечатает в консоле

echo some text.расширение > name - создает файл

git config --global user.name Name - изменить имя автора в коммите

git config --global user.email namemail - поменять почту

git config --global --list - посмотреть имена в коммите

git branch -M main - переименовать ветку
